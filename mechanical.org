    % Controller

    % Reference signal

    \draw[->] (addn.north) -- (addr.south) node[below left]{$d_m$};
    \draw[->] (addr.east) -- (K.west) node[above left]{$\epsilon$};

    \node[addb, right=1 of F] (addF) {};
    \draw[->] (K.east) -| (addF.north);
    \draw[->] (addF.west) -- (F.east) node[above right]{$F$};
